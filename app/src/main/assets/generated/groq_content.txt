7. `MenuModel.java`：菜品模型，用于存储菜品信息

    menuPrice = (TextView) findViewById(R.id.menu_price);
    menuDescription = (TextView) findViewById(R.id.menu_description);
    menuImage = (ImageView) findViewById(R.id.menu_image);
    uploadButton = (Button) findViewById(R.id.upload_button);

    // 设置菜品信息
    MenuModel menuModel = getMenuModel();
    menuName.setText(menuModel.getName());
    menuPrice.setText(String.valueOf(menuModel.getPrice()));
    menuDescription.setText(menuModel.getDescription());

      @Override
